import { DemoShell } from '../../../../components/demo/DemoShell'
import { DataPreview, formatNumber } from '../../../../components/demo/DataPreview'
import influencers from '../../../../data/demo/influencers.json'

export const columns = [
  { key: 'name', label: '이름' },
  { key: 'followers', label: '팔로워', format: formatNumber },
  { key: 'channel', label: '채널' },
  { key: 'category', label: '카테고리' },
  { key: 'tone', label: '톤' },
]

export const prompt = `다음은 보들(BO:DL) K-뷰티 브랜드의 인플루언서 시딩 대상 리스트입니다.

각 인플루언서에게 "2X 프레시 밤투폼" 제품 시딩 이메일을 작성해주세요.

요구사항:
- 각 인플루언서의 채널, 카테고리, 선호 톤에 맞춰 개인화
- 제품 특징: 밤에서 폼으로 변하는 올인원 클렌저, 시어버터+판테놀, 더블클렌징 불필요
- 이메일 구조: 인사 → 브랜드 소개 → 제품 소개 → 협업 제안 → 마무리
- 톤이 "캐주얼"이면 친근하게, "전문적"이면 포멀하게, "따뜻한"이면 감성적으로

인플루언서 리스트 (상위 5명):
${JSON.stringify(influencers.slice(0, 5), null, 2)}

5명의 이메일을 각각 작성해주세요.`

<DemoShell
  title="인플루언서 시딩 이메일 자동화"
  demoNumber={1}
  description="인플루언서 프로필에 맞는 개인화 시딩 이메일을 벌크 생성합니다"
  defaultPrompt={prompt}
  systemPrompt="당신은 K-뷰티 브랜드 보들(BO:DL)의 마케팅 담당자입니다. 인플루언서 시딩 이메일을 작성합니다. 각 인플루언서의 특성에 맞춰 톤과 내용을 개인화하세요."
  dataPreview={<DataPreview data={influencers} columns={columns} maxRows={20} />}
/>
