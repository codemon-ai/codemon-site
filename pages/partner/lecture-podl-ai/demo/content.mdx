import { DemoShell } from '../../../../components/demo/DemoShell'
import { DataPreview } from '../../../../components/demo/DataPreview'
import products from '../../../../data/demo/products.json'

export const columns = [
  { key: 'name', label: '제품명' },
  { key: 'category', label: '카테고리' },
  { key: 'usp', label: 'USP' },
  { key: 'target', label: '타겟' },
]

export const featured = products[0]

export const prompt = `다음은 보들(BO:DL) K-뷰티 브랜드의 제품 정보입니다.

제품: ${featured.name}
카테고리: ${featured.category}
가격: ${featured.price.toLocaleString()}원
성분: ${featured.ingredients.join(', ')}
USP: ${featured.usp}
타겟: ${featured.target}
설명: ${featured.description}

다음 콘텐츠를 모두 생성해주세요:

1. **상세페이지 카피** (500자 내외)
   - 히어로 카피 + 제품 특징 3가지 + CTA

2. **인스타그램 캐러셀 문구 5종**
   - 각 슬라이드 제목 + 본문 (간결하게)
   - 스와이프 유도 문구 포함

3. **해시태그 20개**
   - 브랜드, 제품, 성분, 타겟, 트렌드 키워드 포함

4. **틱톡 숏폼 스크립트** (15초)
   - 후킹 → 제품소개 → CTA 구조

톤: 부드럽고 신뢰감 있게 (보들 브랜드 톤)`

<DemoShell
  title="상품 콘텐츠 자동 생성"
  demoNumber={4}
  description="상세페이지 카피 + SNS 캐러셀 + 해시태그 + 숏폼 스크립트를 동시 생성합니다"
  defaultPrompt={prompt}
  systemPrompt="당신은 보들(BO:DL) K-뷰티 브랜드의 콘텐츠 크리에이터입니다. 제품 정보를 기반으로 다양한 채널용 콘텐츠를 생성합니다. 보들의 브랜드 톤(부드럽고 신뢰감 있는)을 일관되게 유지하세요."
  dataPreview={<DataPreview data={products} columns={columns} />}
/>
